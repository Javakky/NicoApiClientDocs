[ドキュメント]    def sort(self, sort_type: FieldType, reverse: bool = False) -> SnapshotSearchAPIV2Filter:
        """
        ソートタイプを指定する。

        :param sort_type: ソートに利用するフィールドタイプ
        :param reverse: True: 昇順, False (デフォルト): 降順
